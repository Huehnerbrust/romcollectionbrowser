CREATE VIEW ReleaseView AS
       SELECT R.id,
              R.name,
              R.description,
              R.gameCmd,
              R.alternateGameCmd,
              R.mobyRank,
              R.mobyScore,
              R.mobyScoreVotes,
              R.thegamesdbScore,
              R.thegamesdbVotes,
              R.isFavorite,
              R.launchCount,
              R.version,
              R.completed,
              R.broken,
              R.dateAdded,
              R.dateModified,
              R.lastPlayed,
              R.nameFromFile,
              R.firstRomFile,
              R.gameId,
              R.platformId,
              Platform.name AS platform,
              R.developerId,
              Developer.name AS developer,
              R.publisherId,
              Publisher.name AS publisher,
              R.romCollectionId,
              RomCollection.name AS romCollection,
              R.yearId,
              Year.name AS year,
              R.ESRBRatingId,
              ESRBRating.name AS ESRBRating,
              R.regionId,
              Region.name AS region,
              R.languageId,
              Language.name AS language,
              R.mediaId,
              Media.name AS media,
              R.maxPlayersId,
              MaxPlayers.name AS maxPlayers,
              R.perspectiveId,
              Perspective.name AS perspective,
              R.controllerId,
              Controller.name AS controller
         FROM [RELEASE] R
              LEFT OUTER JOIN Platform
                           ON R.platformId = Platform.id
              LEFT OUTER JOIN Company Developer
                           ON R.developerId = Developer.id
              LEFT OUTER JOIN Company Publisher
                           ON R.publisherId = Publisher.id
              LEFT OUTER JOIN RomCollection
                           ON R.romCollectionId = RomCollection.id
              LEFT OUTER JOIN Year
                           ON R.yearId = Year.id
              LEFT OUTER JOIN ESRBRating
                           ON R.ESRBRatingId = ESRBRating.id
              LEFT OUTER JOIN Region
                           ON R.regionId = Region.id
              LEFT OUTER JOIN Language
                           ON R.languageId = Language.id
              LEFT OUTER JOIN Media
                           ON R.mediaId = Media.id
              LEFT OUTER JOIN MaxPlayers
                           ON R.maxPlayersId = MaxPlayers.id
              LEFT OUTER JOIN Perspective
                           ON R.perspectiveId = Perspective.id
              LEFT OUTER JOIN Controller
                           ON R.controllerId = Controller.id;
